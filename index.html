#include #include #include #include #define MAX_RULES 50 #define MAX_PRODS 10 #define MAX_LEN 100 // Grammar storage char nonterminals[MAX_RULES]; char productions[MAX_RULES][MAX_PRODS][MAX_LEN]; int prod_count[MAX_RULES]; int rule_count = 0; char start_symbol; char input[MAX_LEN]; int n; // Function prototypes bool parse(char nonterminal, int *pos); // Utility: find rule index for a nonterminal int find_rule(char nt) { for (int i = 0; i < rule_count; i++) if (nonterminals[i] == nt) return i; return -1; } // Recursive parsing function bool parse(char nt, int *pos) { int rule_idx = find_rule(nt); if (rule_idx == -1) return false; // Try each production for (int p = 0; p < prod_count[rule_idx]; p++) { int save = *pos; // backtrack point bool success = true; for (int k = 0; productions[rule_idx][p][k] != '\0'; k++) { char symbol = productions[rule_idx][p][k]; if (isupper(symbol)) { // Non-terminal if (!parse(symbol, pos)) { success = false; break; } } else { // Terminal if (*pos < n && input[*pos] == symbol) { (*pos)++; } else { success = false; break; } } } if (success) return true; // this production worked *pos = save; // backtrack } return false; } int main() { printf("Enter number of grammar rules: "); scanf("%d", &rule_count); for (int i = 0; i < rule_count; i++) { char lhs; char rhs[MAX_LEN]; printf("Enter rule %d (format: A->xyz|pqr): ", i + 1); scanf("%s", rhs); lhs = rhs[0]; nonterminals[i] = lhs; prod_count[i] = 0; char *token = strtok(rhs + 3, "|"); while (token) { strcpy(productions[i][prod_count[i]++], token); token = strtok(NULL, "|"); } } start_symbol = nonterminals[0]; // assume first rule is start printf("Enter input string: "); scanf("%s", input); n = strlen(input); int pos = 0; if (parse(start_symbol, &pos) && pos == n) { printf("String Accepted!\n"); } else { printf("String Rejected!\n"); } return 0; }
